# 1. Jdi do kořene projektu
cd "f:\AI_Projekt\omni\repository\OmniVoice"

# 2. Inicializace gitu
git init

# 3. Vytvoření .gitignore, aby se nenahrával venv a dočasné soubory
# To je kritické, aby se nenahrávaly gigabajty z venv_win
"venv_win/
__pycache__/
*.pyc
.env
.vscode/" | Out-File -FilePath .gitignore -Encoding utf8

# 4. Přidání všech souborů a první commit
git add .
git commit -m "Initial commit of OmniVoice project"

# 5. Vytvoření repozitáře na GitHubu (použijeme GitHub CLI 'gh' pokud ho máš, 
# jinak vytvoř repozitář ručně na webu a pak pokračuj s těmito příkazy)
# Nahraz 'OmniVoice-Repo' názvem, který chceš, a 'tvuje-jmeno' svým uživatelem
git branch -M main
git remote add origin https://github.com/Brkoszov/OmniVoice-Repo.git
git push -u origin main
